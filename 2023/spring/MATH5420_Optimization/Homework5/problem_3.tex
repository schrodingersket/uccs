Find the dual of 

\begin{mini*}
  {}{z = -2x_1 + 4x_2 - 3x_3}{}{}
  \addConstraint{9x_1 - 2x_2 - 8x_3}{= 5}
  \addConstraint{3x_1 + 3x_2 + 3x_3}{= 7}
  \addConstraint{7x_1 - 5x_2 + 2x_3}{= 9}
  \addConstraint{x_1, x_2, x_3}{\ge 0}.
\end{mini*}

  \ \\

  \begin{mini*}
    {}{w = b^T y = 5y_1 + 7y_2 + 9y_3}{}{}

    \addConstraint{-2y_1 + 3y_2 - 5y_3}{\le  4}
    \addConstraint{-8y_1 - 3y_2 + 2y_3}{\le -3}
    \addConstraint{y_1, y_2, y_3}{\quad \text{unrestricted}}.
  \end{mini*}
